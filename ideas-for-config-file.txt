hooks {
    modify-buffer builtin-syntax-highlight
}

theme other-editor {
    syntax-style {
        keyword #0000ff bold priority:101
        operator #ff0000 underline
    }
}

file .*\.rs {
    syntax {
        keyword \b(let|pub|fn)\b
        operator =|\+|\-|\*
    }

    mode text/command {
        keybinds {
            f set-combo-mode files
            q "prompt, insert hi" # wtf is this shit
        }
    }
}

mode text/edit {
    <esc> set-mode text/command
    else insert-char
}

combo files {
    keybinds {
        s write-buffer
    }
}
